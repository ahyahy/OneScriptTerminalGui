СтильТаблицы.ЧерезСтрочныйЦветСтрок (TableStyle.RowColorThroughLine) Метод

Синтаксис

ЧерезСтрочныйЦветСтрок(<Отступ>, <ЦветоваяСхема>)

Параметры

Отступ (обязательный)
Тип: Число.
Шаг окрашивания строк.
ЦветоваяСхема (обязательный)
Тип: ЦветоваяСхема (ColorScheme).
Цветовая схема, определяющая основной цвет и цвет фона.

Возвращаемое значение

Описание

Окрашивает строки в указанный цвет с шагом в n строк, исходя из переданного в параметре числа.

Примечание

Метод аннулирует действие методов
СтильТаблицы.ЦветДесяткаСтрок (TableStyle.RowColorByDecimal)
СтильТаблицы.ЦветСтроки (TableStyle.RowColor)

Пример

Синий = ТФ.ЦветоваяСхема();
Синий.Нормальный = ТФ.Атрибут(ТФ.Цвет.Серый, ТФ.Цвет.Синий);

СтильТаблицы1.ЧерезСтрочныйЦветСтрок(2, Синий);
Полный пример кода 



Перем ТФ, СтильТаблицы1;

Процедура ПриОткрытии() Экспорт
	Синий = ТФ.ЦветоваяСхема();
	Синий.Нормальный = ТФ.Атрибут(ТФ.Цвет.Серый, ТФ.Цвет.Синий);
	
	СтильТаблицы1.ЧерезСтрочныйЦветСтрок(2, Синий);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Таблица1 = ТФ.Верхний.Добавить(ТФ.Таблица());
Таблица1.Ширина = ТФ.Величина.Абсолютно(80);
Таблица1.Высота = ТФ.Величина.Заполнить();

ТаблицаДанных1 = ТФ.ТаблицаДанных();
Колонки1 = ТаблицаДанных1.Колонки;
Для А = 0 По 100 Цикл
	Колонки1.Добавить(ТФ.КолонкаДанных("Колонка" + А, ТФ.ТипДанных.Строка));
КонецЦикла;
Строки1 = ТаблицаДанных1.Строки;
Для А = 0 По 100 Цикл
	ТекСтрока = Строки1.Добавить(ТаблицаДанных1.НоваяСтрока());
	ТекСтрока.УстановитьЭлемент("Колонка0", А);
	Для А1 = 0 По Колонки1.Количество - 1 Цикл
		ТекСтрока.УстановитьЭлемент("Колонка" + А1, "" + (Число(А) * Число(А1)));
	КонецЦикла;
КонецЦикла;

Таблица1.ТаблицаДанных = ТаблицаДанных1;

СтильТаблицы1 = ТФ.СтильТаблицы();
Таблица1.СтильТаблицы = СтильТаблицы1;

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
