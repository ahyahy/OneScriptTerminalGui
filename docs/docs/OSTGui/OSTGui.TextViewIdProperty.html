Текстовый.Идентификатор (TextView.Id) Свойство

Использование

Чтение и запись.

Значение

Тип: Строка.

Возвращает или задает идентификатор для элемента.

Примечание

Пример

Текстовый1.Идентификатор = "Идент";
Полный пример кода 



Перем ТФ, Текстовый1, Идентификатор, Окно1;

Процедура ПриОткрытии() Экспорт
	Идентификатор = Текстовый1.Идентификатор;
	ТФ.Уведомление().Показать("Идентификатор = " + Идентификатор, 3000);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Окно1 = ТФ.Окно();
Окно1.Заполнить();
ТФ.Верхний.Добавить(Окно1);

Текстовый1 = ТФ.Текстовый();
Текстовый1.Икс = ТФ.Позиция.Абсолютно(2);
Текстовый1.Игрек = ТФ.Позиция.Абсолютно(4);
Текстовый1.Ширина = ТФ.Величина.Абсолютно(50);
Текстовый1.Высота = ТФ.Величина.Абсолютно(10);
Текстовый1.Перенос = Истина;
Текстовый1.Текст = "Текстовый1";
ЦветоваяСхема1 = ТФ.ЦветоваяСхема();
ЦветоваяСхема1.Нормальный = ТФ.Атрибут(ТФ.Цвет.Синий, ТФ.Цвет.Серый);
ЦветоваяСхема1.Фокус = ТФ.Атрибут(ТФ.Цвет.ЯркоГолубой, ТФ.Цвет.ТемноСерый);
Текстовый1.ЦветоваяСхема = ЦветоваяСхема1;
Окно1.Добавить(Текстовый1);

Текстовый1.Идентификатор = "Идент";

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");
ТФ.Запуск();
