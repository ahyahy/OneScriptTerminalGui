Диалог.Подэлементы (Dialog.Subviews) Свойство

Использование

Только чтение.

Значение

Тип: КоллекцияПодэлементов (SubviewCollection).

Возвращает коллекцию подэлементов, содержащую дочерние элементы этого элемента.

Примечание

Пример

Подэлементы1 = Диалог1.Подэлементы;
Полный пример кода 



Перем ТФ, Диалог1;

Процедура ПриОткрытии() Экспорт
	Стр = "Подэлементы для Диалог1:
	|";
	Стр = Стр + "====================
	|Цикл Для ... По ... Цикл
	|";
	Подэлементы1 = Диалог1.Подэлементы;
	Для А = 0 По Подэлементы1.Количество - 1 Цикл
		Стр = Стр + "Элемент(" + А + ") = " + Подэлементы1.Получить(А).Текст + Символы.ПС;
	КонецЦикла;
	Стр = Стр + "====================
	|Цикл Для Каждого ... Из ... Цикл
	|";
	Для Каждого Эл Из Подэлементы1 Цикл
		Стр = Стр + "Эл = " + Эл.Текст + Символы.ПС;
	КонецЦикла;
	
	ТФ.Верхний.Текст = Стр;
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Диалог1 = ТФ.Верхний.Добавить(ТФ.Диалог("Диалог1"));
Диалог1.Икс = ТФ.Позиция.Абсолютно(40);
Диалог1.Игрек = ТФ.Позиция.Абсолютно(10);
Диалог1.СтильКомпоновки = ТФ.СтильКомпоновки.Вычислено;
Диалог1.Ширина = ТФ.Величина.Заполнить(1);
Диалог1.Высота = ТФ.Величина.Заполнить(1);

Кнопка2 = Диалог1.Добавить(ТФ.Кнопка("Кнопка2"));
Кнопка2.Икс = ТФ.Позиция.Абсолютно(3);
Кнопка2.Игрек = ТФ.Позиция.Абсолютно(2);
Граница2 = ТФ.Граница();
Граница2.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка2.Граница = Граница2;

Кнопка3 = Диалог1.Добавить(ТФ.Кнопка("Кнопка3"));
Кнопка3.Игрек = ТФ.Позиция.Абсолютно(2);
Граница3 = ТФ.Граница();
Граница3.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка3.Граница = Граница3;
Кнопка3.Правее(Кнопка2, 2);

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
