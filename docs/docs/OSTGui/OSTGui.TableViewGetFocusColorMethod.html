Таблица.ЦветФокуса (TableView.GetFocusColor) Метод

Синтаксис

ЦветФокуса()

Возвращаемое значение

Тип: Атрибут (Attribute).

Описание

Возвращает текущую цветовую схему для текущего фокуса.

Примечание

Пример

Таблица1.ЦветоваяСхема = ТФ.ЦветоваяСхема();
Таблица1.ЦветоваяСхема.Фокус = ТФ.Атрибут(ТФ.Цвет.Синий, ТФ.Цвет.Серый);
Полный пример кода 



Перем ТФ, Таблица1, ЦветФокуса;

Процедура ПриОткрытии() Экспорт
	ЦветФокуса = Таблица1.ЦветФокуса();
	Стр = "ЦветФокуса() = " + ЦветФокуса + "
	|ОсновнойЦвет = " + ТФ.Цвет.Имя(ЦветФокуса.ОсновнойЦвет) + "
	|ЦветФона = " + ТФ.Цвет.Имя(ЦветФокуса.ЦветФона) + "
	|";
	ТФ.Уведомление().Показать(Стр, -1);
КонецПроцедуры

ТекСтрока.УстановитьЭлемент("№ н/п", 4);
ТекСтрока.УстановитьЭлемент("ФИО", "Петров Петр Петрович");
ТекСтрока = Строки1.Добавить(ТаблицаДанных1.НоваяСтрока());
ТекСтрока.УстановитьЭлемент("№ н/п", 5);
ТекСтрока.УстановитьЭлемент("ФИО", "Сергеевич Аристарх Изяславович");
Таблица1.ТаблицаДанных = ТаблицаДанных1;

Таблица1.ЦветоваяСхема = ТФ.ЦветоваяСхема();

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
