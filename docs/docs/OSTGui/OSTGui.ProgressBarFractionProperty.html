Индикатор.Значение (ProgressBar.Fraction) Свойство

Использование

Чтение и запись.

Значение

Тип: Число.

Возвращает или задает отображаемую долю индикатора, которая должна находиться в диапазоне от 0 до 100.

Примечание

Пример

Значение = Индикатор1.Значение;
Полный пример кода 



Перем ТФ, Индикатор1, Счетчик1, Таймер1;

Процедура ПриОткрытии() Экспорт
	Значение = Индикатор1.Значение;
	Индикатор = Индикатор1;
	ТФ.Верхний.Текст = "Индикатор1.Значение = " + Значение + "
	|Нажмите кнопку Старт";
	ТФ.ОтправитьКлавиши("С", Ложь, Ложь, Ложь);
КонецПроцедуры

Процедура Кнопка1_Нажатие() Экспорт
	Таймер1.Остановить();
	Счетчик1 = 0;
	Индикатор1.Значение = 0;
	Таймер1.Начать();
КонецПроцедуры

Процедура Кнопка2_Нажатие() Экспорт
	Таймер1.Остановить();
КонецПроцедуры

Процедура Таймер1_ПриСрабатыванииТаймера() Экспорт
	Счетчик1 = Счетчик1 + 0.01;
	Индикатор1.Значение = Счетчик1;
	ТФ.Верхний.Текст = "Индикатор1.Значение = " + Индикатор1.Значение;
	Если Счетчик1 >= 1 Тогда
		Индикатор1.Значение = 0.25;
		Таймер1.Остановить();
	КонецЕсли;
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Индикатор1 = ТФ.Верхний.Добавить(ТФ.Индикатор(, , 60));
Индикатор1.Центр();
Индикатор1.Значение = 0.4;
Индикатор1.ЦветоваяСхема = ТФ.Цвета.Основа;
Индикатор1.СтильИндикатора = ТФ.СтильИндикатора.Блоки;

Кнопка1 = ТФ.Верхний.Добавить(ТФ.Кнопка("Старт", , , 10, 1));
Кнопка1.Центр();
Кнопка1.Выше(Индикатор1, 2);
Кнопка1.Граница = ТФ.Граница(ТФ.СтильГраницы.Закругленная);
Кнопка1.Нажатие = ТФ.Действие(ЭтотОбъект, "Кнопка1_Нажатие");

Кнопка2 = ТФ.Верхний.Добавить(ТФ.Кнопка("С_топ", , , 10, 1));
Кнопка2.Центр();
Кнопка2.Ниже(Индикатор1, 2);
Кнопка2.Граница = ТФ.Граница(ТФ.СтильГраницы.Закругленная);
Кнопка2.Нажатие = ТФ.Действие(ЭтотОбъект, "Кнопка2_Нажатие");

Таймер1 = ТФ.Таймер();
Таймер1.Интервал = 80;
Таймер1.ПриСрабатыванииТаймера = ТФ.Действие(ЭтотОбъект, "Таймер1_ПриСрабатыванииТаймера");

Счетчик1 = 0;

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
