Кнопка.ОтключитьКлавишуВызова (Button.DisableHotKey) Свойство

Использование

Чтение и запись.

Значение

Тип: Булево.

Отключает поддержку клавиши вызова для элемента. Значение по умолчанию - Ложь.

Примечание

Пример

Кнопка2.ОтключитьКлавишуВызова = Истина;
Полный пример кода 



ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Кнопка1 = ТФ.Кнопка();
Кнопка1.Текст = "Кн_опка1";
Кнопка1.Икс = ТФ.Позиция.Абсолютно(1);
Кнопка1.Игрек = ТФ.Позиция.Абсолютно(1);
Граница1 = ТФ.Граница();
Граница1.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка1.Граница = Граница1;
ТФ.Верхний.Добавить(Кнопка1);

Кнопка2 = ТФ.Кнопка("К_нопка2");
Кнопка2.Икс = Кнопка1.Икс;
Кнопка2.Ниже(Кнопка1, 1);
Граница2 = ТФ.Граница();
Граница2.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка2.Граница = Граница2;
ТФ.Верхний.Добавить(Кнопка2);

Кнопка2.ОтключитьКлавишуВызова = Истина;

ТФ.Запуск();

Тестовый код 



Перем ОтключитьКлавишуВызова, Кнопка2;

Процедура ПриОткрытии() Экспорт
	ОтключитьКлавишуВызова = Кнопка2.ОтключитьКлавишуВызова;
