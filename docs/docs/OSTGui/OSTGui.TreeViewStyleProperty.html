Дерево.Стиль (TreeView.Style) Свойство

Использование

Чтение и запись.

Значение

Тип: СтильДерева (TreeStyle).

Возвращает или задает параметры для изменения способа отображения дерева.

Примечание

Пример

СтильДерева1 = ТФ.СтильДерева();
СтильДерева1.ПоказатьВертикальные = Ложь;
Дерево1.Стиль = СтильДерева1;
Полный пример кода 



Перем ТФ, Дерево1;

Процедура ПриОткрытии() Экспорт
	СтильДерева1 = ТФ.СтильДерева();
	СтильДерева1.ПоказатьВертикальные = Ложь;
	Дерево1.Стиль = СтильДерева1;
	
	СтильДерева = Дерево1.Стиль;
	ТФ.Уведомление().Показать("СтильДерева = " + СтильДерева, -1);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Дерево1 = ТФ.Дерево(, 6, 40, 15);
Дерево1.ЦветоваяСхема = ТФ.ЦветоваяСхема();
Дерево1.ЦветоваяСхема.Нормальный = ТФ.Атрибут(ТФ.Цвет.Черный, ТФ.Цвет.Серый);
Дерево1.ЦветоваяСхема.Фокус = ТФ.Атрибут(ТФ.Цвет.Синий, ТФ.Цвет.ЯркоГолубой);
ТФ.Верхний.Добавить(Дерево1);

Узел1 = Дерево1.ДобавитьУзел(ТФ.УзелДерева("Узел1"));
Узел2 = Дерево1.ДобавитьУзел(ТФ.УзелДерева("Узел2"));
Узел3 = Дерево1.ДобавитьУзел(ТФ.УзелДерева("Узел3"));

Узел5 = Узел1.Добавить(ТФ.УзелДерева("Узел5"));

Узел7 = Узел6.Добавить(ТФ.УзелДерева("Узел7"));
Узел8 = Узел6.Добавить(ТФ.УзелДерева("Узел8"));

Дерево1.РазвернутьВсе();

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
