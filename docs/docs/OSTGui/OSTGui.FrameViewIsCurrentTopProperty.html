РамкаГруппы.ТекущийСверху (FrameView.IsCurrentTop) Свойство

Использование

Только чтение.

Значение

Тип: Булево.

Возвращает значение, указывающее, является ли этот элемент в настоящее время на переднем плане (активен).

Примечание

Пример

ТекущийСверху = РамкаГруппы1.ТекущийСверху;
Полный пример кода 



Перем ТФ, ТекущийСверху, РамкаГруппы1;

Процедура ПриОткрытии() Экспорт
	ТекущийСверху = РамкаГруппы1.ТекущийСверху;
	ТФ.Уведомление().Показать("ТекущийСверху = " + ТекущийСверху, -1);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

РамкаГруппы1 = ТФ.Верхний.Добавить(ТФ.РамкаГруппы("РамкаГруппы1", 1, 1));
РамкаГруппы1.Ширина = ТФ.Величина.Процент(25);
РамкаГруппы1.Высота = ТФ.Величина.Процент(35);

Кнопка1 = ТФ.Кнопка("Кнопка1", 2, 1);
Кнопка1.Граница = ТФ.Граница();
Кнопка1.Граница.СтильГраницы = ТФ.СтильГраницы.Закругленная;

Кнопка2 = ТФ.Кнопка("Кнопка2", 2);
Кнопка2.Ниже(Кнопка1, 1);
Кнопка2.Граница = Кнопка1.Граница;

РамкаГруппы1.Добавить(Кнопка1);
РамкаГруппы1.Добавить(Кнопка2);

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
