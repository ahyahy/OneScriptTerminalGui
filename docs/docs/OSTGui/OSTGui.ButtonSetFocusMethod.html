Кнопка.УстановитьФокус (Button.SetFocus) Метод

Синтаксис

УстановитьФокус()

Возвращаемое значение

Описание

Устанавливает фокус.

Примечание

Пример

Кнопка3.УстановитьФокус();
Полный пример кода 



Перем ТФ, Кнопка3;

Процедура ПриОткрытии() Экспорт
	Кнопка3.УстановитьФокус();
	ТФ.Верхний.Текст = "Фокус на Кнопка3";
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Окно1 = ТФ.Окно("Окно1");
Окно1.Икс = ТФ.Позиция.Абсолютно(1);
Окно1.Игрек = ТФ.Позиция.Абсолютно(8);
Окно1.Ширина = ТФ.Величина.Абсолютно(50);
Окно1.Высота = ТФ.Величина.Абсолютно(15);
ТФ.Верхний.Добавить(Окно1);

Кнопка1 = ТФ.Кнопка("Кнопка1");
Кнопка1.Икс = ТФ.Позиция.Абсолютно(1);
Кнопка1.Игрек = ТФ.Позиция.Абсолютно(1);
Кнопка1.СтильКомпоновки = ТФ.СтильКомпоновки.Вычислено;
Кнопка1.Ширина = ТФ.Величина.Заполнить(1);
Кнопка1.Высота = ТФ.Величина.Заполнить(1);
Окно1.Добавить(Кнопка1);

Кнопка2 = ТФ.Кнопка("Кнопка2");
Кнопка2.Икс = ТФ.Позиция.Абсолютно(3);
Кнопка2.Игрек = ТФ.Позиция.Абсолютно(2);
Граница2 = ТФ.Граница();
Граница2.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка2.Граница = Граница2;
Кнопка1.Добавить(Кнопка2);

Кнопка3 = ТФ.Кнопка("Кнопка3");
Кнопка3.Игрек = ТФ.Позиция.Абсолютно(2);
Граница3 = ТФ.Граница();
Граница3.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка3.Граница = Граница3;
Кнопка3.Правее(Кнопка2, 2);
Кнопка1.Добавить(Кнопка3);

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
