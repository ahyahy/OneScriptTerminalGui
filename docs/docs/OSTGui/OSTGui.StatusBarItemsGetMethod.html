ЭлементыСтрокиСостояния.Получить (StatusBarItems.Get) Метод

Синтаксис

Получить(<Индекс>)

Параметры

Индекс (обязательный)
Тип: Число.

Возвращаемое значение

Тип: ЭлементСтрокиСостояния (StatusItem).

Описание

Возвращает значение по указанному индексу из коллекции.

Примечание

Пример

ЭлементСтрокиСостояния = СтрокаСостояния1.Элементы.Получить(0);
Полный пример кода 



Перем ТФ, СтрокаСостояния1, ЭлементыСтрокиСостояния;

Процедура ПриОткрытии() Экспорт
	ЭлементыСтрокиСостояния = СтрокаСостояния1.Элементы;
	Стр = "";
	Для А = 0 По ЭлементыСтрокиСостояния.Количество - 1 Цикл
		Стр = Стр + "
		|ЭлементСтрокиСостояния(" + А + ") = " + ЭлементыСтрокиСостояния.Получить(А).Заголовок;
	КонецЦикла;
	
	ТФ.Верхний.Текст = ТФ.Верхний.Текст + "
	|
	|ЭлементыСтрокиСостояния.Количество = " + ЭлементыСтрокиСостояния.Количество + "
	|" + Стр + "
	|";
КонецПроцедуры

Процедура ЭлементСтрокиСостояния1_Нажатие() Экспорт
	ТФ.Верхний.Текст = "
	|Элемент1 Нажатие " + ТекущаяУниверсальнаяДатаВМиллисекундах() + "
	|";
КонецПроцедуры

Процедура ЭлементСтрокиСостояния2_Нажатие() Экспорт
	ТФ.Верхний.Текст = ТФ.Верхний.Текст + "
	|Элемент2 Нажатие " + ТекущаяУниверсальнаяДатаВМиллисекундах() + "
	|";
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

СтрокаСостояния1 = ТФ.СтрокаСостояния();
ТФ.Верхний.Текст = "Кликните по элементам в строке состояния";
ТФ.Верхний.Добавить(СтрокаСостояния1);

ЭлементСтрокиСостояния1 = ТФ.ЭлементСтрокиСостояния(ТФ.Клавиши.s_low, "Элемент1");
ЭлементСтрокиСостояния1.Нажатие = ТФ.Действие(ЭтотОбъект, "ЭлементСтрокиСостояния1_Нажатие");
СтрокаСостояния1.Элементы.Добавить(ЭлементСтрокиСостояния1);

ЭлементСтрокиСостояния2 = ТФ.ЭлементСтрокиСостояния(ТФ.Клавиши.d_low, "Элемент2");
ЭлементСтрокиСостояния2.Нажатие = ТФ.Действие(ЭтотОбъект, "ЭлементСтрокиСостояния2_Нажатие");
СтрокаСостояния1.Элементы.Добавить(ЭлементСтрокиСостояния2);

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
