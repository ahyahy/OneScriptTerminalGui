Дерево.Игрек (TreeView.Y) Свойство

Использование

Чтение и запись.

Значение

Тип: Позиция (Pos).

Возвращает или задает объект Позиция (Pos), соответствующий координате по оси Y.

Примечание

Пример

Дерево1.Игрек = ТФ.Позиция.Абсолютно(3);
Полный пример кода 



Перем ТФ, Дерево1, Игрек;

Процедура ПриОткрытии() Экспорт
	Игрек = Дерево1.Игрек;
	ТФ.Уведомление().Показать("Игрек = " + Игрек, 3000);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Окно1 = ТФ.Окно("Окно1");
Окно1.Заполнить();
ТФ.Верхний.Добавить(Окно1);

Дерево1 = ТФ.Дерево();
Дерево1.Икс = ТФ.Позиция.Абсолютно(1);
Дерево1.Игрек = ТФ.Позиция.Абсолютно(3);
Дерево1.Ширина = ТФ.Величина.Абсолютно(40);
Дерево1.Высота = ТФ.Величина.Абсолютно(15);
ЦветоваяСхема1 = ТФ.ЦветоваяСхема();
ЦветоваяСхема1.Нормальный = ТФ.Атрибут(ТФ.Цвет.Синий, ТФ.Цвет.Серый);
ЦветоваяСхема1.Фокус = ТФ.Атрибут(ТФ.Цвет.ЯркоГолубой, ТФ.Цвет.ТемноСерый);
Дерево1.ЦветоваяСхема = ЦветоваяСхема1;

Узел1 = ТФ.УзелДерева();
Узел1.Текст = "Узел1";
Дерево1.ДобавитьУзел(Узел1);

Узел2 = ТФ.УзелДерева();
Узел2.Текст = "Узел2";
Дерево1.ДобавитьУзел(Узел2);

Узел3 = ТФ.УзелДерева();
Узел3.Текст = "Узел3";
Дерево1.ДобавитьУзел(Узел3);

Узел1_1 = ТФ.УзелДерева();
Узел1_1.Текст = "Узел1_1";
Узел1.Добавить(Узел1_1);

Узел1_2 = ТФ.УзелДерева();
Узел1_2.Текст = "Узел1_2";
Узел1.Добавить(Узел1_2);

Узел1_1_1 = ТФ.УзелДерева();
Узел1_1_1.Текст = "Узел1_1_1";
Узел1_1.Добавить(Узел1_1_1);

Дерево1.РазвернутьВсе();

Окно1.Добавить(Дерево1);

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");
ТФ.Запуск();
