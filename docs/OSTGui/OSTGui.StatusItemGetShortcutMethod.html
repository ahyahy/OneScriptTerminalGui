ЭлементСтрокиСостояния.ПолучитьСочетаниеКлавиш (StatusItem.GetShortcut) Метод

Синтаксис

ПолучитьСочетаниеКлавиш()

Возвращаемое значение

Тип: СписокЗначений.

Список значений из пространства имен односкрипта.

Описание

Возвращает сочетания клавиш, назначенные для элемента.

Примечание

Пример

Список = Объект1.ПолучитьСочетаниеКлавиш();
Полный пример кода 



Перем ТФ, ЭлементСтрокиСостояния1;

Процедура ПриОткрытии() Экспорт
	ЭлементСтрокиСостояния1.УдалитьСочетаниеКлавиш(ТФ.Клавиши.d_low);

	ТФ.ОтправитьКлавиши("s", Ложь, Ложь, Ложь);
КонецПроцедуры

Процедура СочетаниеКлавишДействие() Экспорт
	Объект1 = ТФ.АргументыСобытия.Параметр;
	Список = Объект1.ПолучитьСочетаниеКлавиш();
	Стр = "";
	Для Каждого Эл Из Список Цикл
		Стр = Стр + "
		| " + Эл.Значение + " (" + Эл.Представление + ") " + ТекущаяУниверсальнаяДатаВМиллисекундах();
	КонецЦикла;
	
	ТФ.Верхний.Текст = "
	|
	|
	|
	|
	|
	|
	|
	|СочетаниеКлавиш для " + Объект1.Заголовок + " = " + Стр;
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

СтрокаСостояния1 = ТФ.СтрокаСостояния();
ТФ.Верхний.Добавить(СтрокаСостояния1);

ЭлементСтрокиСостояния1 = ТФ.ЭлементСтрокиСостояния(ТФ.Клавиши.w_low, "Элемент1");
// Нажатие клавиши w для ЭлементСтрокиСостояния1 равносилен клику по нему мышью.
// Только следите за раскладкой (w сработает только при англ. раскладке)
ЭлементСтрокиСостояния1.Нажатие = ТФ.Действие(ЭтотОбъект, "ЭлементСтрокиСостояния1_Нажатие");
СтрокаСостояния1.Элементы.Добавить(ЭлементСтрокиСостояния1);

ЭлементСтрокиСостояния1.ДобавитьСочетаниеКлавиш(ТФ.Клавиши.s_low);
ЭлементСтрокиСостояния1.ДобавитьСочетаниеКлавиш(ТФ.Клавиши.x_low);
ЭлементСтрокиСостояния1.ДобавитьСочетаниеКлавиш(ТФ.Клавиши.d_low);

ЭлементСтрокиСостояния1.СочетаниеКлавишДействие = ТФ.Действие(ЭтотОбъект, "СочетаниеКлавишДействие", ЭлементСтрокиСостояния1);

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
