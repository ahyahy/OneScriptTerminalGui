Индикатор.СтильИндикатора (ProgressBar.ProgressBarStyle) Свойство

Использование

Чтение и запись.

Значение

Тип: СтильИндикатора (ProgressBarStyle).

Возвращает или задает стиль индикатора выполнения.

Примечание

Пример

Индикатор1.СтильИндикатора = ТФ.СтильИндикатора.БегущаяПолоса;
Полный пример кода 



Перем ТФ, Индикатор1, Таймер1;

Процедура ПриОткрытии() Экспорт
	Индикатор = Индикатор1;
	ТФ.Верхний.Текст = "Индикатор1 = " + Индикатор + "
	|Нажмите кнопку Старт";
	ТФ.ОтправитьКлавиши("С", Ложь, Ложь, Ложь);
	СтильИндикатора = Индикатор1.СтильИндикатора;
	ТФ.Уведомление().Показать("СтильИндикатора = " + ТФ.СтильИндикатора.Имя(СтильИндикатора), 2000);
КонецПроцедуры

Процедура Кнопка1_Нажатие() Экспорт
	Таймер1.Остановить();
	Таймер1.Начать();
КонецПроцедуры

Процедура Кнопка2_Нажатие() Экспорт
	Таймер1.Остановить();
КонецПроцедуры

Процедура Таймер1_ПриСрабатыванииТаймера() Экспорт
	Индикатор1.Шаг();
КонецПроцедуры

Индикатор1.Центр();

Кнопка1.Граница = ТФ.Граница(ТФ.СтильГраницы.Закругленная);

Кнопка2.Граница = ТФ.Граница(ТФ.СтильГраницы.Закругленная);
Кнопка2.Нажатие = ТФ.Действие(ЭтотОбъект, "Кнопка2_Нажатие");

Таймер1 = ТФ.Таймер();
Таймер1.Интервал = 80;
Таймер1.ПриСрабатыванииТаймера = ТФ.Действие(ЭтотОбъект, "Таймер1_ПриСрабатыванииТаймера");

Индикатор1.Двунаправленно = Истина;

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
