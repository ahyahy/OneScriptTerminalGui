Кнопка.ПоУмолчанию (Button.IsDefault) Свойство

Использование

Чтение и запись.

Значение

Тип: Булево.

Возвращает или задает значение, указывающее, является ли Кнопка действием по умолчанию для активации в диалоговом окне.

Примечание

Пример

Кнопка5.ПоУмолчанию = Истина;
Полный пример кода 
Копировать     Выделить всё


Перем ПоУмолчанию, Кнопка5, ТФ;

Процедура ПриОткрытии() Экспорт
	ПоУмолчанию = Кнопка5.ПоУмолчанию;
	ТФ.Верхний.Текст = "ПоУмолчанию Кнопка5 = " + ПоУмолчанию;
КонецПроцедуры

Процедура Нажатие() Экспорт
	ТФ.Верхний.Текст = "Нажатие = " + ТекущаяУниверсальнаяДатаВМиллисекундах();
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();
ТФ.Активировать();

Кнопка1.Игрек = ТФ.Позиция.Абсолютно(4);
Кнопка1.Ширина = ТФ.Величина.Абсолютно(25);
Кнопка1.Высота = ТФ.Величина.Абсолютно(3);
Граница1 = ТФ.Граница();
Граница1.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка1.Граница = Граница1;
ТФ.Верхний.Добавить(Кнопка1);

Кнопка2 = ТФ.Кнопка("Кнопка2");

Кнопка2.Икс = Кнопка1.Икс;
Кнопка2.Ниже(Кнопка1, 1);
Граница2 = ТФ.Граница();

Кнопка2.Граница = Граница2;
ТФ.Верхний.Добавить(Кнопка2);

Кнопка3 = ТФ.Кнопка("Кнопка3", Истина);

Кнопка3.Икс = Кнопка1.Икс;
Кнопка3.Ниже(Кнопка2, 1);
Граница3 = ТФ.Граница();
Граница3.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка3.Граница = Граница3;
ТФ.Верхний.Добавить(Кнопка3);

Кнопка4 = ТФ.Кнопка(1, 12, "Кнопка4");
Кнопка4.Ширина = ТФ.Величина.Сложить(Кнопка1.Ширина, ТФ.Величина.Абсолютно(10));
Граница4 = ТФ.Граница();
Граница4.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка4.Граница = Граница4;
ТФ.Верхний.Добавить(Кнопка4);

Кнопка5 = ТФ.Кнопка(1, 15, "Кнопка5", Истина);
Кнопка5.Ширина = ТФ.Величина.Сложить(Кнопка1.Ширина, ТФ.Величина.Абсолютно(10));
Граница5 = ТФ.Граница();
Граница5.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка5.Граница = Граница5;
ТФ.Верхний.Добавить(Кнопка5);

Кнопка5.ПоУмолчанию = Истина;
Кнопка5.Нажатие = ТФ.Действие(ЭтотОбъект, "Нажатие");

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
