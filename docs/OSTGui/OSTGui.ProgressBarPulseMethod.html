Индикатор.Шаг (ProgressBar.Pulse) Метод

Синтаксис

Шаг()

Возвращаемое значение

Описание

Уведомляет индикатор о том, что был достигнут определённый прогресс.

Примечание

Пример

Процедура Таймер1_ПриСрабатыванииТаймера() Экспорт
	Индикатор1.Шаг();
КонецПроцедуры

	Окно1.Текст = "Индикатор1 = " + Индикатор + "

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");

Окно1.Заголовок = "Окно1";
Окно1.Заполнить();

Индикатор1 = Окно1.Добавить(ТФ.Индикатор(, , 60));
Индикатор1.Центр();
Индикатор1.Значение = 0.4;
Индикатор1.ЦветоваяСхема = ТФ.Цвета.Основа;
Индикатор1.СтильИндикатора = ТФ.СтильИндикатора.БегущаяПолоса;

Кнопка1 = Окно1.Добавить(ТФ.Кнопка("Старт", , , 10, 1));
Кнопка1.Центр();
Кнопка1.Выше(Индикатор1, 2);
Кнопка1.Граница = ТФ.Граница(ТФ.СтильГраницы.Закругленная);
Кнопка1.Граница.Эффект3D = Истина;
Кнопка1.Граница.Смещение3D = ТФ.Точка(1, 1);
Кнопка1.Нажатие = ТФ.Действие(ЭтотОбъект, "Кнопка1_Нажатие");

Кнопка2 = Окно1.Добавить(ТФ.Кнопка("С_топ", , , 10, 1));
Кнопка2.Центр();
Кнопка2.Ниже(Индикатор1, 2);
Кнопка2.Граница = ТФ.Граница(ТФ.СтильГраницы.Закругленная);
Кнопка2.Граница.Эффект3D = Истина;
Кнопка2.Граница.Смещение3D = ТФ.Точка(1, 1);
Кнопка2.Нажатие = ТФ.Действие(ЭтотОбъект, "Кнопка2_Нажатие");

Таймер1 = ТФ.Таймер();
Таймер1.Интервал = 80;
Таймер1.ПриСрабатыванииТаймера = ТФ.Действие(ЭтотОбъект, "Таймер1_ПриСрабатыванииТаймера");

Индикатор1.Двунаправленно = Истина;

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
