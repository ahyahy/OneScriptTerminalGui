Дерево.Выбранный (TreeView.SelectedObject) Свойство

Использование

Чтение и запись.

Значение

Тип: УзелДерева (TreeNode).

Возвращает или задает выбранный в данный момент объект в дереве. Если свойство МножественныйВыбор (MultiSelect) имеет значение Истина, то это объект, на котором находится курсор.

Примечание

Пример

Выбранный = Дерево1.Выбранный;
Полный пример кода 



Перем ТФ, Дерево1, Узел1;

Процедура ПриОткрытии() Экспорт
	Дерево1.НавигацияБукв = Истина;
	НавигацияБукв = Дерево1.НавигацияБукв;
	ТФ.ОтправитьКлавиши("з", Ложь, Ложь, Ложь);
	Выбранный = Дерево1.Выбранный;
	ТФ.Уведомление().Показать("Выбранный = " + Выбранный.Текст, -1);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Дерево1 = ТФ.Дерево(, , 40, 15);
Дерево1.ЦветоваяСхема = ТФ.ЦветоваяСхема();
Дерево1.ЦветоваяСхема.Нормальный = ТФ.Атрибут(ТФ.Цвет.Черный, ТФ.Цвет.Серый);
Дерево1.ЦветоваяСхема.Фокус = ТФ.Атрибут(ТФ.Цвет.Синий, ТФ.Цвет.ЯркоГолубой);

Белый = Дерево1.ДобавитьУзел(ТФ.УзелДерева("Белый"));
Голубой = Дерево1.ДобавитьУзел(ТФ.УзелДерева("Голубой"));
Зеленый = Дерево1.ДобавитьУзел(ТФ.УзелДерева("Зеленый"));
Красный = Белый.Добавить(ТФ.УзелДерева("Красный"));
Пурпурный = Белый.Добавить(ТФ.УзелДерева("Пурпурный"));
Синий = Белый.Добавить(ТФ.УзелДерева("Синий"));
ТемноСерый = Синий.Добавить(ТФ.УзелДерева("ТемноСерый"));
Черный = Синий.Добавить(ТФ.УзелДерева("Черный"));

Дерево1.РазвернутьВсе();

ТФ.Верхний.Добавить(Дерево1);

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
