ПанельМеню.Добавлено (MenuBar.IsAdded) Свойство

Использование

Только чтение.

Значение

Тип: Булево.

Возвращает информацию о том добавлен ли элемент к родителю.

Примечание

Пример

Добавлено = ПанельМеню1.Добавлено;
Полный пример кода 
Копировать     Выделить всё


Перем ТФ, ПанельМеню1, Текст;

Процедура ПриОткрытии() Экспорт
	Добавлено = ПанельМеню1.Добавлено;
	ТФ.Верхний.Текст = ТФ.Верхний.Текст + "
	|Добавлено = " + Добавлено + "
	|";
КонецПроцедуры

Процедура Файл_Нажатие() Экспорт

КонецПроцедуры

Процедура Справка_Нажатие() Экспорт

КонецПроцедуры

Процедура Открыть_Нажатие() Экспорт

КонецПроцедуры

КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();
ТФ.Активировать();

ПанельМеню1 = ТФ.ПанельМеню();

Структура1 = Новый Структура("Планета","Земля");
Структура1.Вставить("Звезда", "Солнце");
Структура1.Вставить("Спутник", "Луна");
ПанельМеню1.Данные = Структура1;

ТФ.Верхний.Добавить(ПанельМеню1);

ПунктМеню1 = ТФ.ПунктМеню();
ПунктМеню1.Заголовок = "Файл";
ПунктМеню1.Нажатие = ТФ.Действие(ЭтотОбъект, "Файл_Нажатие");
ПанельМеню1.ПодМеню.Добавить(ПунктМеню1);

ЭлементМеню1 = ТФ.ЭлементМеню();
ЭлементМеню1.Заголовок = "_Открыть";
ЭлементМеню1.Нажатие = ТФ.Действие(ЭтотОбъект, "Открыть_Нажатие");
ПунктМеню1.Элементы.Добавить(ЭлементМеню1);

ЭлементМеню2 = ТФ.ЭлементМеню("_Выход", "Выход");
ЭлементМеню2.Нажатие = ТФ.Действие(ЭтотОбъект, "Выход_Нажатие");
ПунктМеню1.Элементы.Добавить(ЭлементМеню2);

ПунктМеню2 = ТФ.ПунктМеню("Справка", "Справка");
ПунктМеню2.Нажатие = ТФ.Действие(ЭтотОбъект, "Справка_Нажатие");
ПанельМеню1.ПодМеню.Добавить(ПунктМеню2);

Текст = "
|
|
|
|
|
|
|Кликните по элементам панели меню.
|";
ТФ.Верхний.Текст = Текст;

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
