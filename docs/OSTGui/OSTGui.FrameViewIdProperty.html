РамкаГруппы.Идентификатор (FrameView.Id) Свойство

Использование

Чтение и запись.

Значение

Тип: Строка.

Возвращает или задает идентификатор для элемента.

Примечание

Пример

РамкаГруппы1.Идентификатор = "Идент";
Полный пример кода 



Перем ТФ, РамкаГруппы1, Идентификатор, Окно1;

Процедура ПриОткрытии() Экспорт
	Идентификатор = РамкаГруппы1.Идентификатор;
	ТФ.Уведомление().Показать("Идентификатор = " + Идентификатор, 3000);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Окно1 = ТФ.Окно();
Окно1.Заполнить();
ТФ.Верхний.Добавить(Окно1);

РамкаГруппы1 = ТФ.РамкаГруппы();
РамкаГруппы1.Заголовок = "РамкаГруппы1";
РамкаГруппы1.Икс = ТФ.Позиция.Абсолютно(1);
РамкаГруппы1.Игрек = ТФ.Позиция.Абсолютно(1);
РамкаГруппы1.СтильКомпоновки = ТФ.СтильКомпоновки.Вычислено;
РамкаГруппы1.Ширина = ТФ.Величина.Процент(25);
РамкаГруппы1.Высота = ТФ.Величина.Процент(35);

Кнопка1 = ТФ.Кнопка();
Кнопка1.Текст = "Кнопка1";
Кнопка1.Икс = ТФ.Позиция.Абсолютно(2);
Кнопка1.Игрек = ТФ.Позиция.Абсолютно(1);
Граница1 = ТФ.Граница();
Граница1.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка1.Граница = Граница1;

Кнопка2 = ТФ.Кнопка();
Кнопка2.Текст = "Кнопка2";
Кнопка2.Икс = ТФ.Позиция.Абсолютно(2);
Кнопка2.Ниже(Кнопка1, 1);
Граница2 = ТФ.Граница();
Граница2.СтильГраницы = ТФ.СтильГраницы.Закругленная;
Кнопка2.Граница = Граница2;

РамкаГруппы1.Добавить(Кнопка1);
РамкаГруппы1.Добавить(Кнопка2);

Окно1.Добавить(РамкаГруппы1);

РамкаГруппы1.Идентификатор = "Идент";

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");
ТФ.Запуск();
