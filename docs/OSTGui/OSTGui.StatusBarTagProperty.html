СтрокаСостояния.Метка (StatusBar.Tag) Свойство

Использование

Чтение и запись.

Значение

Тип: Произвольный.

Возвращает или задает объект, содержащий дополнительные данные.

Примечание

Пример

Структура1 = Новый Структура("Буратино","Деревянная кукла, наивная, добродушная и озорная.");
Структура1.Вставить("Карло", "Шарманщик, который вырезал Буратино из полена.");
Структура1.Вставить("Дуремар", "Продавец лечебных пиявок и друг Карабаса.");
Структура1.Вставить("Тортила", "Добрая черепаха, которая подарила Буратино золотой ключик.");
СтрокаСостояния1.Метка = Структура1;
Полный пример кода 



Перем ТФ, СтрокаСостояния1;

Процедура ПриОткрытии() Экспорт
	Метка = СтрокаСостояния1.Метка;
	Текст = "
	|
	|
	|
	|
	|
	|
	|СтрокаСостояния1.Метка = " + Метка + "
	|";
	ТФ.Верхний.Текст = Текст;
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

СтрокаСостояния1 = ТФ.СтрокаСостояния();
ТФ.Верхний.Добавить(СтрокаСостояния1);

ЭлементСтрокиСостояния1 = ТФ.ЭлементСтрокиСостояния(ТФ.Клавиши.s_low, "Элемент1");
// Нажатие клавиши s для ЭлементСтрокиСостояния1 равносилен клику по нему мышью.
// Только следите за раскладкой (s сработает только при англ. раскладке)
ЭлементСтрокиСостояния1.Нажатие = ТФ.Действие(ЭтотОбъект, "ЭлементСтрокиСостояния1_Нажатие");
СтрокаСостояния1.Элементы.Добавить(ЭлементСтрокиСостояния1);

Структура1 = Новый Структура("Буратино","Деревянная кукла, наивная, добродушная и озорная.");
Структура1.Вставить("Карло", "Шарманщик, который вырезал Буратино из полена.");
Структура1.Вставить("Дуремар", "Продавец лечебных пиявок и друг Карабаса.");
Структура1.Вставить("Тортила", "Добрая черепаха, которая подарила Буратино золотой ключик.");
СтрокаСостояния1.Метка = Структура1;

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");

ТФ.Запуск();
