Переключатель.Идентификатор (RadioGroup.Id) Свойство

Использование

Чтение и запись.

Значение

Тип: Строка.

Возвращает или задает идентификатор для элемента.

Примечание

Пример

Переключатель1.Идентификатор = "Идент";
Полный пример кода 



Перем ТФ, Переключатель1, Идентификатор, Окно1;

Процедура ПриОткрытии() Экспорт
	Идентификатор = Переключатель1.Идентификатор;
	ТФ.Уведомление().Показать("Идентификатор = " + Идентификатор, 3000);
КонецПроцедуры

ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Окно1 = ТФ.Окно();
Окно1.Заполнить();
ТФ.Верхний.Добавить(Окно1);

Переключатель1 = ТФ.Переключатель();
Переключатель1.Икс = ТФ.Позиция.Абсолютно(1);
Переключатель1.Игрек = ТФ.Позиция.Абсолютно(2);

Метки1 = Переключатель1.Метки;
Метки1.Добавить("Метка1");
Метки1.Добавить("Метка2");
Метки1.Добавить("Метка3");

Переключатель1.ИндексВыбранного = 1;
Окно1.Добавить(Переключатель1);

Переключатель1.Идентификатор = "Идент";

ТФ.ПриОткрытии = ТФ.Действие(ЭтотОбъект, "ПриОткрытии");
ТФ.Запуск();
