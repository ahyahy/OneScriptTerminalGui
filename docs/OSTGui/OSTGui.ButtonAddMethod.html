Кнопка.Добавить (Button.Add) Метод

Синтаксис

Добавить(<Элемент>)

Параметры

Элемент (обязательный)
Тип: Произвольный.

Возвращаемое значение

Описание

Добавляет дочерний элемент к этому элементу.

Примечание

Пример

Кнопка1.Добавить(Кнопка2);
Полный пример кода 



ПодключитьВнешнююКомпоненту("C:\444\111\OneScriptTerminalGui\OneScriptTerminalGui\bin\Debug\OneScriptTerminalGui.dll");
ТФ = Новый ТерминалФормыДляОдноСкрипта();

Кнопка1 = ТФ.Кнопка();
Кнопка1.Икс = ТФ.Позиция.Центр();
Кнопка1.Игрек = ТФ.Позиция.Центр();
Кнопка1.Ширина = ТФ.Величина.Процент(60);
Кнопка1.Высота = ТФ.Величина.Процент(60);
Кнопка1.Текст = "Кнопка1";
ТФ.Верхний.Добавить(Кнопка1);

Кнопка2 = ТФ.Кнопка(2, 2, 15, 1);
Кнопка2.Текст = "Кнопка2";
Кнопка2.Граница = ТФ.Граница();
Кнопка2.Граница.СтильГраницы = ТФ.СтильГраницы.Одинарная;
Кнопка1.Добавить(Кнопка2);

Кнопка3 = ТФ.Кнопка(2, 2, 15, 1);
Кнопка3.Текст = "Кнопка3";
Кнопка3.Граница = ТФ.Граница();
Кнопка3.Граница.СтильГраницы = ТФ.СтильГраницы.Одинарная;
Кнопка3.Правее(Кнопка2, 2);
Кнопка1.Добавить(Кнопка3);

ТФ.Запуск();
